      #Maximum Continuous Days of Trending
      def find_max_period(dates):
          period=1
          max_periods = 1
          for index, date in enumerate(dates[:-1]):
              diff = dates[index+1] - date
              
              if diff.days == 1:
                  period+=1
              else:
                  if max_periods < period:
                      max_periods = period
                  period=1
              
          return (max_periods if max_periods>period else period)
